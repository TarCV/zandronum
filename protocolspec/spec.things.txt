Command MoveThing
	Actor actor
	Short bits

	If (bits & CM_X)
		AproxFixed newX
	EndIf

	If (bits & CM_Y)
		AproxFixed newY
	EndIf

	If (bits & CM_Z)
		AproxFixed newZ
	EndIf

	If (bits & CM_LAST_X)
		AproxFixed lastX
	EndIf

	If (bits & CM_LAST_Y)
		AproxFixed lastY
	EndIf

	If (bits & CM_LAST_Z)
		AproxFixed lastZ
	EndIf

	If (bits & CM_ANGLE)
		Angle angle
	EndIf

	If (bits & CM_VELX)
		AproxFixed velX
	EndIf

	If (bits & CM_VELY)
		AproxFixed velY
	EndIf

	If (bits & CM_VELZ)
		AproxFixed velZ
	EndIf

	If (bits & CM_PITCH)
		Long pitch
	EndIf

	If (bits & CM_MOVEDIR)
		Byte movedir
	EndIf
EndCommand
